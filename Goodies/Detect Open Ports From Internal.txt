To Detect all open ports:

      #!/bin/bash
      port=1
      while [ $port -le 65535 ]
      do
      nc -z -v  portquiz.net $port
      ((port++))
      done

2.) chmod 777 open_ports.sh
3.) run --> ./open_ports.sh
